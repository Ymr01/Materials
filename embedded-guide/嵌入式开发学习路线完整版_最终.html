1.1 指针的本质与进阶用法

指针是嵌入式开发的核心，必须深刻理解其内存模型。

指针与数组的区别

// 数组名是常量指针，不可修改
int arr[5] = {1, 2, 3, 4, 5};
int *p = arr;  // arr退化为指针

// 关键区别
sizeof(arr);   // 20字节（5个int）
sizeof(p);     // 4或8字节（指针本身大小）

// 数组名不可修改
// arr++;  // 错误！
p++;       // 正确

#define BIT_SET(reg, bit)     ((reg) |= (1U << (bit)))
#define BIT_CLR(reg, bit)     ((reg) &= ~(1U << (bit)))
#define BIT_TOGGLE(reg, bit)  ((reg) ^= (1U << (bit)))
#define BIT_READ(reg, bit)    (((reg) >> (bit)) & 1U)

// 实际应用：配置STM32 GPIO

BIT_CLR(GPIOA_ODR, 5);    // PA5输出低电平

通信报文位域打包与解包

嵌入式通信中常见的位操作场景：

// 自定义协议：传感器状态包
// Byte0-1: 数值（0.1单位，范围0-6553.5）
// Byte2 bit0-3: 模式（0=待机, 1=运行, 2=故障, 3=校准）
// Byte2 bit4: 使能状态

void PackSensorStatus(uint8_t *buf, float value, uint8_t mode, bool enabled) {
    uint16_t raw = (uint16_t)(value * 10);
    buf[0] = (raw >> 8) & 0xFF;
    buf[1] = raw & 0xFF;

    buf[2] = (mode & 0x0F);
    if (enabled) {
        buf[2] |= (1 << 4);
    }
}

void UnpackSensorStatus(uint8_t *buf, float *value, uint8_t *mode, bool *enabled) {
    uint16_t raw = (buf[0] << 8) | buf[1];
    *value = raw * 0.1f;
    *mode = buf[2] & 0x0F;
    *enabled = (buf[2] >> 4) & 0x01;
}

模板与类型安全

template<typename T, size_t N>
class RingBuffer {
private:
    T buffer[N];
    size_t head = 0, tail = 0;
public:
    bool Push(const T &item) {
        size_t next = (head + 1) % N;
        if (next == tail) return false;
        buffer[head] = item;
        head = next;
        return true;
    }
    bool Pop(T &item) {
        if (head == tail) return false;
        item = buffer[tail];
        tail = (tail + 1) % N;
        return true;
    }
};

2.3 时钟树配置

// 配置系统时钟到168MHz（STM32F4）
void SystemClock_Config(void) {
    RCC->CR |= RCC_CR_HSEON;
    while (!(RCC->CR & RCC_CR_HSERDY));

    // VCO = HSE × (PLLN / PLLM) = 8 × (336 / 8) = 336MHz
    // SYSCLK = VCO / PLLP = 168MHz
    RCC->PLLCFGR = (8 << 0) | (336 << 6) | (0 << 16) | (RCC_PLLCFGR_PLLSRC_HSE);

    RCC->CR |= RCC_CR_PLLON;
    while (!(RCC->CR & RCC_CR_PLLRDY));

    RCC->CFGR |= RCC_CFGR_HPRE_DIV1;
    RCC->CFGR |= RCC_CFGR_PPRE1_DIV4;
    RCC->CFGR |= RCC_CFGR_PPRE2_DIV2;

    RCC->CFGR |= RCC_CFGR_SW_PLL;
    while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL);
}

3.1 UART串口通信

DMA方式收发（推荐）

uint8_t uart_tx_buf[256];
volatile bool uart_tx_busy = false;

void UART_DMA_Init(void) {
    RCC->AHB1ENR |= RCC_AHB1ENR_DMA1EN;
    DMA1_Stream6->CR = (4 << 25) | (1 << 10) | (1 << 6) | (0 << 8);
    DMA1_Stream6->PAR = (uint32_t)&USART2->DR;
    DMA1_Stream6->M0AR = (uint32_t)uart_tx_buf;
}

bool UART_SendDMA(const char *str, uint16_t len) {
    if (uart_tx_busy) return false;
    memcpy(uart_tx_buf, str, len);
    DMA1_Stream6->NDTR = len;
    DMA1_Stream6->CR |= DMA_SxCR_EN;
    uart_tx_busy = true;
    return true;
}

void DMA1_Stream6_IRQHandler(void) {
    if (DMA1->HISR & DMA_HISR_TCIF6) {
        DMA1->HIFCR |= DMA_HIFCR_CTCIF6;
        uart_tx_busy = false;
    }
}

void SPI1_Init(void) {
    RCC->APB2ENR |= RCC_APB2ENR_SPI1EN;
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;
    GPIOA->MODER |= (2 << 10) | (2 << 12) | (2 << 14);
    GPIOA->AFR[0] |= (5 << 20) | (5 << 24) | (5 << 28);

uint8_t SPI_ReadWriteByte(uint8_t data) {
    while (!(SPI1->SR & SPI_SR_TXE));
    SPI1->DR = data;
    while (!(SPI1->SR & SPI_SR_RXNE));
    return SPI1->DR;
}

uint8_t MPU6050_ReadReg(uint8_t reg) {
    CS_LOW();
    SPI_ReadWriteByte(reg | 0x80);
    uint8_t value = SPI_ReadWriteByte(0xFF);
    CS_HIGH();
    return value;
}

3.3 I2C总线

软件模拟I2C（兼容性好）

#define SCL_HIGH()  GPIOB->BSRR = (1 << 6)
#define SCL_LOW()   GPIOB->BSRR = (1 << 22)
#define SDA_HIGH()  GPIOB->BSRR = (1 << 7)
#define SDA_LOW()   GPIOB->BSRR = (1 << 23)
#define SDA_READ()  (GPIOB->IDR & (1 << 7))

void I2C_Start(void) {
    SDA_HIGH(); SCL_HIGH();
    I2C_Delay();
    SDA_LOW();
    I2C_Delay();
    SCL_LOW();
}

void I2C_Stop(void) {
    SDA_LOW(); SCL_HIGH();
    I2C_Delay();
    SDA_HIGH();
    I2C_Delay();
}

bool I2C_SendByte(uint8_t data) {
    for (int i = 0; i < 8; i++) {
        if (data & 0x80) SDA_HIGH();
        else SDA_LOW();
        I2C_Delay();
        SCL_HIGH(); I2C_Delay();
        SCL_LOW();
        data <<= 1;
    }
    SDA_HIGH(); I2C_Delay();
    SCL_HIGH(); I2C_Delay();
    bool ack = !SDA_READ();
    SCL_LOW();
    return ack;
}

struct tcp_pcb *server_pcb;

err_t Server_Accept(void *arg, struct tcp_pcb *newpcb, err_t err) {
    tcp_recv(newpcb, Server_Recv);
    return ERR_OK;
}

err_t Server_Recv(void *arg, struct tcp_pcb *tpcb, struct pbuf *p, err_t err) {
    if (p == NULL) {
        tcp_close(tpcb);
        return ERR_OK;
    }
    uint8_t *data = (uint8_t *)p->payload;
    ProcessData(data, p->len);
    tcp_recved(tpcb, p->len);
    pbuf_free(p);
    return ERR_OK;
}

void Server_Init(void) {
    server_pcb = tcp_new();
    tcp_bind(server_pcb, IP_ADDR_ANY, 8080);
    server_pcb = tcp_listen(server_pcb);
    tcp_accept(server_pcb, Server_Accept);
}

    BaseType_t xHigherPriorityTaskWoken = pdFALSE;

    Message_t msg;

        }
    }
}

all: $(TARGET).bin

%.o: %.c

$(TARGET).elf: $(OBJECTS)
	$(CC) $(CFLAGS) -T$(LDSCRIPT) $^ -o $@
	$(SIZE) $@

$(TARGET).bin: $(TARGET).elf
	$(OBJCOPY) -O binary $< $@

clean:

void Watchdog_Init(void) {

    IWDG->PR = 6;       // 预分频64
    IWDG->RLR = 625;    // 重载值
    IWDG->KR = 0xCCCC;  // 启动
}

void MainLoop(void) {
    for (;;) {
        Task1();
        Task2();
        IWDG->KR = 0xAAAA;  // 喂狗
        HAL_Delay(100);
    }
}

// 窗口看门狗（WWDG）- 检测任务超时
void WindowWatchdog_Init(void) {
    WWDG->CFR = 0x7F | (3 << 7);
    WWDG->CR = 0x7F | WWDG_CR_WDGA;
}

6.3 低功耗设计

电池供电的IoT/可穿戴设备必须考虑功耗管理：

  • Standby模式：仅备份域工作 (~nA级)

    // 唤醒后需要重新配置时钟
    SystemClock_Config();
}

6.4 Bootloader与IAP升级

产品发布后通常需要支持OTA升级。典型双区方案：

// Flash分区
// 0x08000000 - 0x08008000  Bootloader (32KB)
// 0x08008000 - 0x08040000  Application A (224KB)
// 0x08040000 - 0x08078000  Application B / OTA缓冲区
// 0x08078000 - 0x08080000  参数区

void Bootloader_Main(void) {
    // 1. 读取启动标志
    uint32_t boot_flag = *(uint32_t *)PARAM_ADDR;

    if (boot_flag == OTA_PENDING) {
        // 2. 校验OTA区固件
        if (Verify_Firmware(OTA_ADDR) == OK) {
            // 3. 拷贝到App区
            Flash_Copy(OTA_ADDR, APP_ADDR, FIRMWARE_SIZE);
            *(uint32_t *)PARAM_ADDR = OTA_DONE;
        }
    }

    // 4. 跳转到App
    JumpToApp(APP_ADDR);
}

    for (int i = 0; i < 128; i++) input->data.f[i] = data[i];
    interpreter->Invoke();
    TfLiteTensor* output = interpreter->output(0);
    int max_idx = 0;

    }

}

11.1 C语言基础

Q1：指针和数组的区别？

int arr[5] = {1, 2, 3, 4, 5};
int *p = arr;

// sizeof不同
printf("%zu\n", sizeof(arr));  // 20字节
printf("%zu\n", sizeof(p));    // 4/8字节

// 取地址不同
printf("%p\n", arr);           // 数组首元素地址
printf("%p\n", &arr);          // 整个数组地址（数值相同，类型不同）
printf("%p\n", &p);            // 指针变量自身地址

// 作为函数参数
void func(int a[]) {  // 等价于 int *a
    printf("%zu\n", sizeof(a));  // 4/8字节（指针）
}

Q2：volatile作用？不用会怎样？

// 硬件寄存器访问
#define GPIO_IDR  (*(volatile uint32_t *)0x40020010)

// 如果不加volatile，编译器可能优化成：
uint32_t temp = GPIO_IDR;
while (temp == 0);  // 死循环！

// 加volatile后，每次都从硬件读取
while (GPIO_IDR == 0);  // 正常工作

Q3：如何定义只读数据？

// 1. const修饰（可能在RAM中）
const int data1 = 100;

// 2. 强制放Flash（GCC）
const int data2 __attribute__((section(".rodata"))) = 200;

// 3. 指针指向Flash

11.2 硬件相关

Q4：如何配置GPIO为推挽输出？

void GPIO_InitOutput(void) {
    // 1. 使能时钟
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;

    // 2. 配置模式（01=输出）
    GPIOA->MODER &= ~(3 << 10);
    GPIOA->MODER |= (1 << 10);

    // 3. 配置输出类型（0=推挽）
    GPIOA->OTYPER &= ~(1 << 5);

    // 4. 配置速度（10=高速50MHz）
    GPIOA->OSPEEDR |= (2 << 10);

    // 5. 配置上下拉（00=无）
    GPIOA->PUPDR &= ~(3 << 10);
}

Q5：中断优先级如何配置？

// ARM Cortex-M优先级规则
// - 数字越小，优先级越高
// - 抢占优先级高可以打断正在执行的中断

NVIC_SetPriorityGrouping(5);

// USART中断：抢占优先级1，响应优先级2
NVIC_SetPriority(USART1_IRQn, NVIC_EncodePriority(5, 1, 2));

// Timer中断：抢占优先级2，响应优先级1
NVIC_SetPriority(TIM2_IRQn, NVIC_EncodePriority(5, 2, 1));

// 结果：USART可以打断Timer，但Timer不能打断USART

11.3 通信协议

Q6：SPI和I2C有什么区别？

特性SPII2C
线数4根（SCK/MOSI/MISO/CS）2根（SCL/SDA）

Q7：UART通信中如何防止数据丢失？

  • 硬件FIFO：利用MCU自带的接收FIFO（一般4-16字节）
  • DMA接收：配置循环DMA，减少中断次数
  • 环形缓冲区：中断中快速存入缓冲，任务中慢速处理
  • 流控：RTS/CTS硬件流控，或XON/XOFF软件流控

Q8：任务切换的时机？

// 1. 时间片耗尽（Tick中断）
void SysTick_Handler(void) {
    if (xTaskIncrementTick() != pdFALSE) {
        portYIELD();  // 触发PendSV切换

Q9：如何避免优先级反转？

// 问题场景：
// 低优先级任务L持有互斥锁
// 高优先级任务H等待该锁
// 中优先级任务M抢占L，导致H被间接阻塞

// 解决方案：优先级继承（Mutex自动支持）
SemaphoreHandle_t mutex = xSemaphoreCreateMutex();

// 任务L获取锁后，优先级临时提升到H的级别
xSemaphoreTake(mutex, portMAX_DELAY);
// ... 临界区代码
xSemaphoreGive(mutex);  // 优先级恢复

// 超声波测距
float Ultrasonic_GetDistance(void) {
    TRIG_HIGH();
    delay_us(10);
    TRIG_LOW();

    uint32_t timeout = 10000;
    while (ECHO_READ() == 0 && timeout--);

    uint32_t start = TIM2->CNT;
    timeout = 50000;
    while (ECHO_READ() == 1 && timeout--);
    uint32_t end = TIM2->CNT;

    float distance = (end - start) * 0.017f;  // cm
    return distance;
}

// 避障逻辑
void ObstacleAvoidance(void) {
    float distance = Ultrasonic_GetDistance();

    if (distance < 20.0f) {
        Motor_Stop();
        delay_ms(100);
        Motor_Backward(50);
        delay_ms(500);
        Motor_TurnRight(50);
        delay_ms(300);
    } else {
        Motor_Forward(70);
    }
}

// FreeRTOS任务架构
void Task_Sensor(void *pvParameters) {
    while (1) {
        float temp = DHT22_ReadTemp();
        float humi = DHT22_ReadHumi();

        // 发送到队列
        SensorData_t data = {temp, humi, millis()};
        xQueueSend(sensor_queue, &data, 0);

        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

void Task_Upload(void *pvParameters) {
    SensorData_t data;
    while (1) {
        if (xQueueReceive(sensor_queue, &data, portMAX_DELAY)) {
            // MQTT上传阿里云
            char payload[128];
            sprintf(payload, "{\"temp\":%.1f,\"humi\":%.1f}", data.temp, data.humi);
            mqtt_publish("sensor/data", payload);

            // SD卡存储
            SD_WriteLog(&data);

            // 超限报警
            if (data.temp > 35.0f) {
                Buzzer_On();
            }
        }
    }
}
